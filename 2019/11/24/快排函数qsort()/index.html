快排函数qsort()

//快排函数qsort()

#include<stdio.h>
#include<stdlib.h>

int *cmp(const int *a, const int * b)
{
int *a = *_a;
int *b = *_b;
return *a - *b;
}
int main()
{
int mass[100];
int num = 100;
int size = sizeof(int);
qsort(mass, num, size, cmp);
}
